def prompt(self): while True: cmd = input(f'\n{self.addr[0]}@{self.client["node"]}>> ') if cmd == ':help': self.help() elif cmd == ':info': print('\nClient Machine Information:\n' + self.client['table'].table) else: try: self.conn.send(cmd.encode()) self.log.info(f'Command Sent: {cmd}', ['file']) if cmd in [':kill', ':exit']: return else: result = self.conn.recv(16834).decode('utf-8', errors='ignore') # Decode with 'ignore' errors print('\n' + result) except socket.error as error: self.log.error(f'Socket Send Error: {error}') return
